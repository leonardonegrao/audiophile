# NumberField

The `NumberField` allows an user to increment or decrement a given value.

export const Template = (args) => {
  const [value, setValue] = useState(args.value);
  console.log(value);
  return (
    <NumberField
      value={value}
      onChange={(value) => {
        setValue(value);
      }}
      {...args}
    />
  );
};
